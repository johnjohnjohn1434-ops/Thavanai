<script>
    // Database Storage
    let seetuData = JSON.parse(localStorage.getItem('seetuDB_v3')) || [];
    
    // Secret verification string (hidden from display ui)
    const secureKey = "thivashthivash143@gmail.com";

    function checkUser() {
        let inputVal = document.getElementById('username').value.trim();
        
        if (inputVal === "") {
            alert("Please input a valid user identity!");
            return;
        }

        // 1. Admin Authentication Pathway
        if (inputVal.toLowerCase() === secureKey.toLowerCase()) {
            document.getElementById('login-section').style.display = 'none';
            document.getElementById('admin-panel').style.display = 'block';
            document.getElementById('user-view').style.display = 'none';
            renderAdminTable();
        } 
        // 2. Client Home Page Dashboard Pathway
        else {
            let user = seetuData.find(u => u.name.toLowerCase() === inputVal.toLowerCase());
            
            document.getElementById('login-section').style.display = 'none';
            document.getElementById('user-view').style.display = 'block';
            document.getElementById('admin-panel').style.display = 'none';
            
            let welcome = document.getElementById('welcome-msg');
            let detailsDiv = document.getElementById('seetu-details');

            if (user) {
                welcome.innerHTML = `Vanakkam, <strong>${user.name}</strong>`;
                
                let perWeekAmt = user.amount / user.totalWeeks;
                let finalPaid = user.paid * perWeekAmt;
                let finalBalance = user.amount - finalPaid;
                let remWeeks = user.totalWeeks - user.paid;
                
                detailsDiv.innerHTML = `
                    <div class="dashboard-card">
                        <div class="grid-stats">
                            <div class="stat-box"><span>Motha Seetu Thogai</span><strong style="color: #a855f7;">₹${user.amount}</strong></div>
                            <div class="stat-box"><span>Motha Thavanai</span><strong>${user.totalWeeks} Weeks</strong></div>
                            <div class="stat-box"><span>Katriya Weeks</span><strong style="color: var(--success);">${user.paid}</strong></div>
                            <div class="stat-box"><span>Meedha Weeks</span><strong>${remWeeks}</strong></div>
                            <div class="stat-box" style="background: rgba(16, 185, 129, 0.1);"><span>Katriya Total Kaasu</span><strong style="color: var(--success);">₹${finalPaid.toFixed(2)}</strong></div>
                            <div class="stat-box" style="background: rgba(239, 68, 68, 0.1);"><span>Meedham Katta Vendiyathu</span><strong style="color: var(--danger);">₹${finalBalance.toFixed(2)}</strong></div>
                        </div>
                    </div>
                `;
            } else {
                welcome.innerHTML = `Welcome Guest!`;
                detailsDiv.innerHTML = `
                    <div class="dashboard-card" style="text-align:center;">
                        <p style="color: var(--danger); font-weight:600;">⚠️ Intha Name-la account illa ba!</p>
                        <p style="font-size:14px; color:var(--text-muted);">Intha user portal-la register panna unga local seetu admin-ah contact pannunga.</p>
                    </div>`;
            }
        }
    }

    function addMember() {
        let name = document.getElementById('new-name').value.trim();
        let amount = parseFloat(document.getElementById('seetu-amount').value);
        let totalWeeks = parseInt(document.getElementById('total-installments').value);

        if (name === "" || isNaN(amount) || isNaN(totalWeeks)) {
            alert("Please completely fill out all parameters.");
            return;
        }

        if (seetuData.some(u => u.name.toLowerCase() === name.toLowerCase())) {
            alert("Member already registered in systemic database!");
            return;
        }

        seetuData.push({ name: name, amount: amount, totalWeeks: totalWeeks, paid: 0 });
        saveData();
        renderAdminTable();
        document.getElementById('new-name').value = "";
    }

    function renderAdminTable() {
        let tbody = document.getElementById('admin-table-body');
        tbody.innerHTML = "";

        if (seetuData.length === 0) {
            tbody.innerHTML = `<tr><td colspan="7" style="text-align:center; color: var(--text-muted); padding:20px;">No database logs identified.</td></tr>`;
            return;
        }

        seetuData.forEach((user, index) => {
            let stepAmt = user.amount / user.totalWeeks;
            let currentPaid = user.paid * stepAmt;
            let currentBal = user.amount - currentPaid;

            let tr = document.createElement('tr');
            tr.innerHTML = `
                <td><input type="text" class="table-input name-input" id="m-name-${index}" value="${user.name}"></td>
                <td><input type="number" class="table-input" id="m-amt-${index}" value="${user.amount}"></td>
                <td><input type="number" class="table-input" id="m-tot-${index}" value="${user.totalWeeks}"></td>
                <td><input type="number" class="table-input" id="m-paid-${index}" value="${user.paid}"></td>
                <td><span style="color: var(--success); font-weight:600;">₹${currentPaid.toFixed(1)}</span></td>
                <td><span style="color: var(--danger); font-weight:600;">₹${currentBal.toFixed(1)}</span></td>
                <td>
                    <div class="action-btns">
                        <button class="btn-success" onclick="executeInlineSave(${index})">Save</button>
                        <button class="btn-danger" onclick="removeMember(${index})">Del</button>
                    </div>
                </td>
            `;
            tbody.appendChild(tr);
        });
    }

    function executeInlineSave(index) {
        let name = document.getElementById(`m-name-${index}`).value.trim();
        let amt = parseFloat(document.getElementById(`m-amt-${index}`).value);
        let tot = parseInt(document.getElementById(`m-tot-${index}`).value);
        let paid = parseInt(document.getElementById(`m-paid-${index}`).value);

        if(name === "" || isNaN(amt) || isNaN(tot) || isNaN(paid)) {
            alert("Empty structural values rejected.");
            return;
        }

        if(paid > tot) {
            alert("Paid intervals cannot exceed maximum allocation!");
            return;
        }

        seetuData[index] = { name: name, amount: amt, totalWeeks: tot, paid: paid };
        saveData();
        renderAdminTable();
    }

    function removeMember(index) {
        if (confirm("Are you sure you want to completely drop this ledger dataset?")) {
            seetuData.splice(index, 1);
            saveData();
            renderAdminTable();
        }
    }

    function saveData() {
        localStorage.setItem('seetuDB_v3', JSON.stringify(seetuData));
    }

    function logout() {
        document.getElementById('username').value = "";
        document.getElementById('login-section').style.display = 'block';
        document.getElementById('admin-panel').style.display = 'none';
        document.getElementById('user-view').style.display = 'none';
    }
